private bool Straight()
        {
            bool sequentalRank = true;
            // Second loop is there to check if five-high straight
            for (int j  = 1; j < 3; j++)
            {
                var sortedHands = Player.Hands.OrderBy(p => p.CardName).ToList();

                for (int i = 0; i < sortedHands.Count() - 1; i++)
                {
                    if ((Convert.ToInt32(sortedHands[i + 1].CardName) - Convert.ToInt32(sortedHands[i].CardName)) != 1)
                    {
                        sequentalRank = false;
                        break;
                    }
                }
                var aceCard = Player.Hands.Where(p => p.CardName == (int)CardRanks.A).FirstOrDefault();
                
                if (aceCard != null && !sequentalRank && (aceCard.CardName == (int)CardRanks.A))
                {
                    Player.Hands.Where(p => p.CardName == aceCard.CardName).First().CardName = 1;
                    sequentalRank = true;
                } 
            }

            return sequentalRank;
        }
