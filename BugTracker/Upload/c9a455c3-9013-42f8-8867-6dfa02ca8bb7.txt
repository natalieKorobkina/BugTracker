[Authorize(Roles = "Admin, ProjectManager")]
        public ActionResult AllTickets()
        {
            //var temp = DbContext.Tickets.Select(p => p).ToList();
            var model = DbContext.Tickets.Select(p => new AllTicketsViewModel
            {
                Id = p.Id,
                ProjectName = DbContext.Projects.Where(pp =>pp.Id == p.ProjectId).FirstOrDefault().Name,
                Title = p.Title,
                DateCreated = p.DateCreated,
                DateUpdated = p.DateUpdated,
                Type = DbContext.TicketTypes.Where(t =>t.Id == p.TicketTypeId).FirstOrDefault().Name,
                Status = DbContext.TicketStatuses.Where(t => t.Id == p.TicketStatusId).FirstOrDefault().Name,
                Priority = DbContext.TicketPriorities.Where(t => t.Id == p.TicketPriorityId).FirstOrDefault().Name,
                Author = DbContext.Users.Where(t => t.Id == p.OwnerUserId).FirstOrDefault().DisplayName,
                Assignee = DbContext.Users.Where(t => t.Id == p.AssignedToUserId).FirstOrDefault().DisplayName

            }).ToList();

            return View(model);
        }
